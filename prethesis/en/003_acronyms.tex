\section*{Glossary}
\begin{acronym}
	\acro{AMD}{Advanced Micro Devices}
	\acro{BLAS}{Basic Linear Algebra Subprograms}
	\acro{CPU}{Central Processing Unit}
	\acro{CUDA}{formerly Compute Unified Device Architecture, a parallel computing platform for \nvidia \acs{GPU}}
	\acro{DSP}{Digital Signal Processor}
	\acro{GAMA}{\gpu And Multi-core Aware}{, an API for shared memory multiprocessing in platforms with a host \acs{CPU} and an attached \cuda enabled accelerator.}
	\acro{GPU}{Graphics Processing Unit}
	\acro{GPGPU}{General Purpose \acs{GPU}}
	\acro{HetPlat}{Heterogeneous Platform}
	\acro{HPC}{High Performance Computing}
	\acro{LAPACK}{Linear Algebra PACKage}
	\acro{MAGMA}{Matrix Algebra on GPU and Multi-core Architectures}
	\acro{MKL}{Math Kernel Library}
	\acro{MIC}{Many Integrated Core}
	\acro{NAG}{Numerical Algorithm Group}
	\acro{PCI}{Peripheral Component Interconnect}
	\acro{PCIe}{\pci Express}
	\acro{SFU}{Special Function Unit}
	\acro{SIMD}{Single Instruction Multiple Data}
	\acro{SIMT}{Single Instruction, Multiple Threads}{, the term used by \nvidia to describe their \acs{GPU} architectures.}
	\acro{SM}{Streaming Multiprocessor}
	\acro{SMX}{Kepler Streaming Multiprocessor. A redesign of the original \acs{SM}.}
	\acro{TBB}{Threading Building Blocks}
	\acro{UMinho}{University of Minho}
\end{acronym}
